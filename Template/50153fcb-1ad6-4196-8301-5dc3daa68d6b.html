--> 测试数据：[tbl]
go
if object_id('[tbl]') is not null
drop table [tbl]
go
create table [tbl](
[母料编号] varchar(3),
[子料编号] varchar(3),
[数量] int
)
go
insert [tbl]
select 'a01','b01',2 union all
select 'a01','b02',1 union all
select 'b01','c01',2 union all
select 'b01','c02',1 union all
select 'c01','d01',2

;with t
as
( select 1 '层数',母料编号,子料编号,数量,数量 '累计数量'
from tbl a
where not exists(
select 1 from tbl b where b.子料编号=a.母料编号)
union all
select t.层数+1,c.母料编号,c.子料编号,
c.数量,c.数量*t.累计数量
from tbl c inner join t
on c.母料编号=t.子料编号
)
select * from t

/*
层数 母料编号 子料编号 数量 累计数量
1 a01 b01 2 2
1 a01 b02 1 1
2 b01 c01 2 4
2 b01 c02 1 2
3 c01 d01 2 8
*/
