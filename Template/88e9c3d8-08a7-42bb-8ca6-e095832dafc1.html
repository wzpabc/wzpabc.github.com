已知表名查数据库名


uploaded by Patrick. 06/27/2016 15:17:42

declare @Table table(DB sysname,TabName sysname)
insert @Table
exec sp_msforeachdb 'select ''?'' as DB,Name as 表名 from [?]..sysobjects where type=''U'' and Name =''表名'''--輸入表名

select * from @Table
