declare @s varchar(50)
set @s='Level 1-2,Tower E5,Plaza,No.44 East Chang An Ave. 东长安街44号东方广场东5座1-2层'
while PATINDEX('%[吖-做]%',@s)>0
set @s=stuff(@s,PATINDEX('%[吖-做]%',@s),1,'')
select @s
