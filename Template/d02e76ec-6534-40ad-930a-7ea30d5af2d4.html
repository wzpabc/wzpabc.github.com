GO
IF OBJECT_ID('TBL')IS NOT NULL
DROP TABLE TBL
GO
CREATE TABLE TBL(
编码 VARCHAR(10),
名称 VARCHAR(10),
颜色 VARCHAR(10),
S INT ,
M INT ,
L INT,
XL INT ,
XXL INT
)
GO
INSERT TBL
SELECT '100000','牛仔裤','蓝色',1,1,1,1,0 UNION ALL
SELECT '100001','衬衣','白色',1,0,0,0,1
select * from tbl

SELECT * FROM(
select * from tbl
unpivot ([数量] for [尺码] in([S],[M],[L],[XL],[XXL]))b)A
WHERE [数量]<>0
/*
编码 名称 颜色 数量 尺码
100000 牛仔裤 蓝色 1 S
100000 牛仔裤 蓝色 1 M
100000 牛仔裤 蓝色 1 L
100000 牛仔裤 蓝色 1 XL
100001 衬衣 白色 1 S
100001 衬衣 白色 1 XXL
*/
